//docker-compose.yml (für Windows)
//Alle Daten sind dabei auf "c:\ProgramData\" gespeichert.

version: "3.8"

services:
  n8n:
    image: n8nio/n8n
    container_name: n8n
    restart: always
    ports:
      - "5678:5678"
    environment:
      - DB_TYPE=mssql
      - DB_MSSQLDB_HOST=sqlserver
      - DB_MSSQLDB_PORT=1433
      - DB_MSSQLDB_DATABASE=n8n
      - DB_MSSQLDB_USER=sa
      - DB_MSSQLDB_PASSWORD=India#2026
      - N8N_BASIC_AUTH_ACTIVE=true
      - N8N_BASIC_AUTH_USER=admin
      - N8N_BASIC_AUTH_PASSWORD=admin123
      - N8N_DEFAULT_BINARY_DATA_MODE=filesystem
    depends_on:
      - sqlserver
    volumes:
      - "C:/ProgramData/n8n/n8n_data:/home/node/.n8n"
    networks:
      - n8n_net

  sqlserver:
    image: mcr.microsoft.com/mssql/server:2022-latest
    container_name: sqlserver
    restart: always
    environment:
      - ACCEPT_EULA=Y
      - SA_PASSWORD=India#2026
    ports:
      - "1433:1433"
    volumes:
      - "C:/ProgramData/n8n/sql_data:/var/opt/mssql"
    networks:
      - n8n_net

  mosquitto:
    image: eclipse-mosquitto:latest
    container_name: mosquitto
    restart: always
    ports:
      - "1883:1883"
      - "9001:9001"
    volumes:
      - "C:/ProgramData/n8n/mosquitto/mosquitto.conf:/mosquitto/config/mosquitto.conf"
      - "C:/ProgramData/n8n/mosquitto/mosquitto_data:/mosquitto/data"
      - "C:/ProgramData/n8n/mosquitto/mosquitto_log:/mosquitto/log"
    networks:
      - n8n_net

  web:
    image: nginx:alpine
    container_name: web
    restart: always
    ports:
      - "8080:80"
    volumes:
      - "C:/ProgramData/n8n/web:/usr/share/nginx/html"
    networks:
      - n8n_net

networks:
  n8n_net:
    driver: bridge
