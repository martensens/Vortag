<script>
const client = mqtt.connect("ws://192.168.0.5:9001");

client.on("connect", () => {
  console.log("MQTT verbunden");
  client.subscribe("customer/result");
});

client.on("message", (topic, message) => {
  const result = message.toString();

  if (result === "OK") {
    document.getElementById("result").innerHTML = "✅ Zahlung erlaubt";
    document.getElementById("result").style.color = "lime";
  } else {
    document.getElementById("result").innerHTML = "❌ Nicht genug Guthaben";
    document.getElementById("result").style.color = "red";
  }
});

function send() {
  const kunde = document.getElementById("kunde").value;
  const betrag = document.getElementById("betrag").value;

  const payload = kunde + "," + betrag;

  client.publish("customer/check", payload);
}
</script>
