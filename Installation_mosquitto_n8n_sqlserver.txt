//Das *.yml File, dass Container für mosquitto, n8n und sqlserver baut. Inklusive Volumes zur persistenten Sicherung und eigenem Netzwerk
//
nano docker-compose.yml
version: "3.8"

services:
  n8n:
    image: n8nio/n8n
    container_name: n8n
    restart: always
    ports:
      - "5678:5678"
    environment:
      - DB_TYPE=mssql
      - DB_MSSQLDB_HOST=sqlserver
      - DB_MSSQLDB_PORT=1433
      - DB_MSSQLDB_DATABASE=n8n
      - DB_MSSQLDB_USER=sa
      - DB_MSSQLDB_PASSWORD=India#2026
      - N8N_BASIC_AUTH_ACTIVE=true
      - N8N_BASIC_AUTH_USER=admin
      - N8N_BASIC_AUTH_PASSWORD=admin123
    depends_on:
      - sqlserver
    volumes:
      - n8n_data:/home/node/.n8n
    networks:
      - n8n_net

  sqlserver:
    image: mcr.microsoft.com/mssql/server:2022-latest
    container_name: sqlserver
    restart: always
    environment:
      - ACCEPT_EULA=Y
      - SA_PASSWORD=India#2026
    ports:
      - "1433:1433"
    volumes:
      - sql_data:/var/opt/mssql
    networks:
      - n8n_net

  mosquitto:
    image: eclipse-mosquitto:latest
    container_name: mosquitto
    restart: always
    ports:
      - "1883:1883"
      - "9001:9001"
    volumes:
      - ./mosquitto/mosquitto.conf:/mosquitto/config/mosquitto.conf
      - mosquitto_data:/mosquitto/data
      - mosquitto_log:/mosquitto/log
    networks:
      - n8n_net

web:
  image: nginx:alpine
  container_name: web
  restart: always
  ports:
    - "8080:80"
  volumes:
    - ./web:/usr/share/nginx/html
  networks:
    - n8n_net

volumes:
  n8n_data:
  sql_data:
  mosquitto_data:
  mosquitto_log:

networks:
  n8n_net:
    driver: bridge

//Speichern und schließen
>-und speichern/schließen

//Starten des docker-compose.yml
docker compose up -d
